What is Turtle?

Understand the concept of turtle graphics and why it's the perfect first graphics library.

Python's turtle module is a built-in library that lets you draw shapes, patterns, and artwork by controlling a small cursor — the "turtle" — on screen. It traces a path as it moves.

Origin: Turtle graphics originated in the Logo programming language in the 1960s, designed to teach children programming through drawing.

How it works

Imagine a turtle holding a pen on paper. You give it commands:

  • Move forward — the turtle walks forward and draws a line
  • Turn left/right — the turtle changes direction
  • Pen up/down — lift pen to move without drawing
  • Change color — set the pen or fill color
hello_turtle.py 
import turtle          # Import the module

t = turtle.Turtle()    # Create a turtle object
t.forward(100)        # Move forward 100 pixels
t.left(90)           # Turn left 90 degrees
t.forward(100)        # Move forward again

turtle.done()         # Keep window open

The coordinate system

The turtle starts at the center (0, 0) of the screen, facing right (east). Coordinates increase going right and up, decrease going left and down.

Your First Steps

Learn the essential movement and pen commands to draw anything.

Movement Commands

movement.py
import turtle
t = turtle.Turtle()

# Movement
t.forward(100)   # or t.fd(100)
t.backward(50)  # or t.bk(50)
t.right(90)      # or t.rt(90) — turn clockwise
t.left(45)       # or t.lt(45) — turn counter-clockwise

# Teleport (no drawing)
t.penup()        # lift pen
t.goto(200, 100) # jump to coordinate
t.pendown()      # put pen down

# Reset
t.home()         # go back to (0,0)
t.clear()        # erase drawings
t.reset()        # clear + go home

Drawing your first square

square.py
import turtle
t = turtle.Turtle()

# Draw a square — 4 sides, 4 right-angle turns
for i in range(4):
    t.forward(100)
    t.right(90)

turtle.done()
Formula: For any regular polygon with n sides, the turn angle is 360 / n degrees. Square = 360/4 = 90°. Triangle = 360/3 = 120°.

Shapes & Loops

Use for-loops to draw complex patterns with very little code.

The real power of turtle graphics comes when you combine loops with movement. A loop that repeats movement + turn can create stunning patterns.

Star

star.py
import turtle
t = turtle.Turtle()
t.speed(10)   # 1=slow, 10=fast, 0=instant

for i in range(5):
    t.forward(150)
    t.right(144)  # 360/5 * 2 = 144 for a star

turtle.done()

    t.right(60)

Reusable Functions

Organize turtle code into functions for cleaner, reusable drawing routines.

Once you need to draw the same shape multiple times, functions are essential. A function lets you define drawing logic once and call it anywhere with different parameters.

functions.py
import turtle
t = turtle.Turtle()
t.speed(0)

def draw_polygon(sides, size, color="white"):
    """Draw a regular polygon at the turtle's current position."""
    t.pencolor(color)
    angle = 360 / sides
    for _ in range(sides):
        t.forward(size)
        t.right(angle)

def move_to(x, y):
    """Move without drawing."""
    t.penup()
    t.goto(x, y)
    t.pendown()

# Use the functions
colors = ["red", "gold", "cyan", "magenta"]
positions = [(-150, 0), (-50, 0), (50, 0), (150, 0)]

for i, (x, y) in enumerate(positions):
    move_to(x, y)
    draw_polygon(i + 3, 40, colors[i])

turtle.done()

Screen & Events

Make your turtle respond to keyboard and mouse input for interactive programs.

events.py
import turtle

s = turtle.Screen()
s.title("My Turtle App")
s.setup(width=600, height=500)
s.bgcolor("black")

t = turtle.Turtle()
t.color("lime")
t.speed(0)

# Keyboard event handlers
def move_forward():  t.forward(20)
def move_back():     t.backward(20)
def turn_left():     t.left(15)
def turn_right():    t.right(15)
def pen_toggle():
    if t.isdown(): t.penup()
    else: t.pendown()

# Bind keys
s.onkey(move_forward, "Up")
s.onkey(move_back,    "Down")
s.onkey(turn_left,   "Left")
s.onkey(turn_right,  "Right")
s.onkey(pen_toggle,  "space")
s.listen()  # start listening for input

# Mouse click: move turtle there
s.onclick(lambda x, y: t.goto(x, y))

turtle.done()
Tip: Always call s.listen() after binding keys, or the screen won't capture keyboard input.

Stamps & Shapes

Use stamps to leave copies of the turtle, and change turtle appearance.

stamps.py
import turtle
t = turtle.Turtle()
t.speed(0)

# Turtle shapes: arrow, turtle, circle, square, triangle, classic
t.shape("turtle")
t.shapesize(stretch_wid=2, stretch_len=2)

# Stamp leaves a permanent copy at current position
colors = ["red", "orange", "yellow", "green"]
for i in range(20):
    t.fillcolor(colors[i % len(colors)])
    t.stamp()          # leave a copy here
    t.forward(30)
    t.right(18)        # makes a circle of turtles

# Custom shape with polygon
turtle.register_shape("diamond", 
    ((0,10), (10,0), (0,-10), (-10,0)))

Recursive Trees

Use recursion to create organic branching structures like trees and fractals.

Recursion means a function calls itself. With turtle graphics, this creates breathtaking self-similar structures called fractals.

fractal_tree.py
import turtle
t = turtle.Turtle()
t.speed(0)
t.left(90)           # point upward
t.penup()
t.goto(0, -200)      # start near bottom
t.pendown()

def draw_branch(length, depth):
    if depth == 0:     # base case — stop recursing
        return
    
    # Color: brown for trunk, green for leaves
    if length < 20:
        t.pencolor("green")
    else:
        t.pencolor("saddlebrown")
    
    t.pensize(depth)    # thicker for trunk
    t.forward(length)  # draw this branch
    
    t.right(25)         # turn right, draw right subtree
    draw_branch(length * 0.7, depth - 1)
    
    t.left(50)          # turn left, draw left subtree
    draw_branch(length * 0.7, depth - 1)
    
    t.right(25)         # restore original direction
    t.backward(length) # back up the branch

draw_branch(100, 8)
turtle.done()
Key insight: Every recursive call reduces the depth by 1. When depth == 0, the function stops — this is the base case that prevents infinite loops.

Lindenmayer Systems

Generate fractal patterns using string-rewriting rules — the Koch Snowflake and more.

lsystem.py
import turtle

def apply_rules(axiom, rules, iterations):
    """Expand string using L-system rules."""
    result = axiom
    for _ in range(iterations):
        result = "".join(rules.get(c, c) for c in result)
    return result

def draw_lsystem(t, instructions, angle, length):
    """Interpret L-system string as turtle commands."""
    stack = []
    for cmd in instructions:
        if   cmd == "F": t.forward(length)
        elif cmd == "+": t.right(angle)
        elif cmd == "-": t.left(angle)
        elif cmd == "[": stack.append((t.pos(), t.heading()))
        elif cmd == "]":
            pos, head = stack.pop()
            t.penup(); t.goto(pos); t.setheading(head); t.pendown()

# Koch Snowflake
rules = {"F": "F+F--F+F"}
axiom = "F--F--F"
instructions = apply_rules(axiom, rules, 4)
draw_lsystem(t, instructions, angle=60, length=5)

# Plant-like L-System
rules = {"F": "FF", "X": "F+[-F-XF-X][+FF][--XF[+X]][++F-X]"}
axiom = "X"
instructions = apply_rules(axiom, rules, 4)
draw_lsystem(t, instructions, angle=25, length=5)

OOP with Multiple Turtles

Use classes and multiple turtle instances to build complex animated scenes.

oop_turtles.py
import turtle
import random

class Firework:
    """A turtle that draws a firework burst."""
    
    def __init__(self, x, y, color, rays=12):
        self.t = turtle.Turtle()
        self.t.speed(0)
        self.t.penup()
        self.t.goto(x, y)
        self.t.pendown()
        self.t.color(color)
        self.t.hideturtle()
        self.rays = rays
    
    def explode(self, size=50):
        angle = 360 / self.rays
        for _ in range(self.rays):
            self.t.forward(size)
            self.t.backward(size)
            self.t.left(angle)
    
    def clear(self):
        self.t.clear()

# Create multiple fireworks
s = turtle.Screen()
s.bgcolor("black")

fireworks = [
    Firework(-150, 100, "red"),
    Firework(0, 50, "gold", rays=8),
    Firework(150, 100, "cyan", rays=16),
]

for fw in fireworks:
    fw.explode(80)

turtle.done()
